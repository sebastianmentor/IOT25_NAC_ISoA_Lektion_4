MCU     = atmega328p
F_CPU   = 16000000UL
TARGET  = app
BUILD_DIR = build

CC      = avr-gcc
OBJCOPY = avr-objcopy
SIZE    = avr-size
AVRDUDE = avrdude

PROGRAMMER = arduino
PORT       = /dev/tty.usbserial-0001   # ändra efter din enhet (t.ex. /dev/tty.usbmodemXXXX)
BAUD       = 115200

CFLAGS = -mmcu=$(MCU) -DF_CPU=$(F_CPU) -Os -Wall -Wextra -std=c11 \
         -Iinclude -Ihal -Idrivers/gpio -Idrivers/uart -Isrc -Idrivers/millis -Idrivers/keypad -Idrivers/i2c

SRCS = \
	src/main.c \
	src/app.c \
	drivers/gpio/gpio.c \
	drivers/uart/uart.c \
	drivers/uart/ring_buffer.c\
	drivers/millis/millis.c\
	drivers/keypad/keypad.c\
	drivers/i2c/twi.c

OBJS = $(SRCS:%.c=$(BUILD_DIR)/%.o)
ELF  = $(BUILD_DIR)/$(TARGET).elf
HEX  = $(BUILD_DIR)/$(TARGET).hex
MAP  = $(BUILD_DIR)/$(TARGET).map

all: $(HEX)

$(HEX): $(ELF)
	$(OBJCOPY) -O ihex -R .eeprom $< $@
	$(SIZE) $<

$(ELF): $(OBJS)
	$(CC) $(CFLAGS) -Wl,-Map=$(MAP) -o $@ $^

$(BUILD_DIR)/%.o: %.c
	@mkdir -p $(dir $@)
	$(CC) $(CFLAGS) -c $< -o $@

flash: $(HEX)
	$(AVRDUDE) -c $(PROGRAMMER) -p m328p -P $(PORT) -b $(BAUD) -U flash:w:$(HEX):i

clean:
	rm -rf $(BUILD_DIR)

.PHONY: all clean flash
